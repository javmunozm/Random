// Kino Data Extractor - Run this in browser console on each series page

function extractKinoData() {
    // Try multiple extraction strategies

    // Strategy 1: Look for number elements with common class names
    let numbers = [];
    const numberElements = document.querySelectorAll('[class*="ball"], [class*="numero"], [class*="number"], [class*="bola"]');

    numberElements.forEach(el => {
        const text = el.textContent.trim();
        const match = text.match(/\b(0[1-9]|1[0-9]|2[0-5])\b/);
        if (match) {
            numbers.push(parseInt(match[1]));
        }
    });

    // Strategy 2: If no luck, extract all numbers from page text
    if (numbers.length < 98) {
        numbers = [];
        const bodyText = document.body.innerText;
        const matches = bodyText.match(/\b(0[1-9]|1[0-9]|2[0-5])\b/g);
        if (matches) {
            numbers = matches.map(n => parseInt(n));
        }
    }

    // Group into 7 events of 14 numbers each
    const events = [];
    for (let i = 0; i < Math.min(7, Math.floor(numbers.length / 14)); i++) {
        const event = numbers.slice(i * 14, (i + 1) * 14).sort((a, b) => a - b);
        if (event.length === 14 && new Set(event).size === 14) {
            events.push(event);
        }
    }

    if (events.length === 7) {
        console.log('✅ Successfully extracted 7 events!');
        console.log(JSON.stringify(events, null, 2));
        return events;
    } else {
        console.log('⚠️ Only found', events.length, 'complete events');
        console.log('Total numbers found:', numbers.length);
        return null;
    }
}

// Run extraction
const result = extractKinoData();

// Copy to clipboard if successful
if (result) {
    navigator.clipboard.writeText(JSON.stringify(result, null, 2));
    console.log('📋 Data copied to clipboard!');
}
